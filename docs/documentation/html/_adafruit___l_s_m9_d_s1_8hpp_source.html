201  class Sensor: public Adafruit_Sensor {

210  Sensor(Adafruit_LSM9DS1* parent, lsm9ds1_read_func readFunc,
211  lsm9ds1_get_event_func eventFunc, lsm9ds1_get_sensor_func sensorFunc):
212  _parent(parent),
213  _readFunc(readFunc),
214  _eventFunc(eventFunc),
215  _sensorFunc(sensorFunc)
216  {}
217  virtual bool getEvent(sensors_event_t* event) {
218  /* Take new reading. */
219  (_parent->*_readFunc)();
220  /* Fill in event data. */
221  (_parent->*_eventFunc)(event, 0);
222  }
223  virtual void getSensor(sensor_t* sensor) {
224  /* Fill in sensor metadata. */
225  (_parent->*_sensorFunc)(sensor);
226  }
