291  inline uint8_t reverse(uint8_t b) {
292  b = (b & 0xF0) >> 4 | (b & 0x0F) << 4;
293  b = (b & 0xCC) >> 2 | (b & 0x33) << 2;
294  b = (b & 0xAA) >> 1 | (b & 0x55) << 1;
295  return b;
296  }
